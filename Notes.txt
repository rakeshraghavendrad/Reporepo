from deepeval.metrics import FaithfulnessMetric
from deepeval.test_case import LLMTestCase
import re

# âœ… Custom FaithfulnessMetric to enforce JSON response format
class PatchedFaithfulnessMetric(FaithfulnessMetric):
    def _construct_prompt(self, context, output):
        return f"""
Evaluate how faithful the following output is to the given context.

Return ONLY in this exact JSON format:
{{
  "score": float (between 0 and 1),
  "reason": string
}}

Context:
"{context}"

Output:
"{output}"
"""

# âœ… Your LLM output and context
actual_output = "We offer a 30-day full refund at no extra cost."
retrieval_context = ["All customers are eligible for a 30 day full refund at no extra cost."]

# âœ… Use the patched FaithfulnessMetric
metric = PatchedFaithfulnessMetric(
    threshold=0.7,
    model="gemini-2.0-flash-001",  # or "gpt-4"
    include_reason=True
)

# âœ… Define your test case
test_case = LLMTestCase(
    input="What if these shoes don't fit?",
    actual_output=actual_output,
    retrieval_context=retrieval_context
)

# âœ… Run evaluation and handle JSON + non-JSON responses
try:
    metric.measure(test_case)
    print("âœ… Score:", getattr(metric, 'score', 'â“ Not Set'))
    print("ğŸ’¬ Reason:", getattr(metric, 'reason', 'â“ Not Set'))

except Exception as e:
    print("âŒ Model output could not be parsed as JSON.")
    print("ğŸ” Error:", e)

    # Get raw model output from the error
    raw_response = e.args[0] if hasattr(e, 'args') and len(e.args) > 0 else None
    if raw_response:
        print("ğŸ“¤ Raw model response:\n", raw_response)

        # âœ… Extract score using regex
        score_match = re.search(r'score\s*(is|of|=)?\s*([0-9]\.?[0-9]*)', raw_response, re.IGNORECASE)
        score = float(score_match.group(2)) if score_match else "â“ Not found"

        # âœ… Extract reason (text before 'score')
        reason = raw_response.split("score")[0].strip().strip('"')

        print("\nğŸ” Extracted values (from freeform text):")
        print("âœ… Score:", score)
        print("ğŸ’¬ Reason:", reason)
