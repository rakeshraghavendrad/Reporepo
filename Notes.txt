for keyword, trigger_patterns in kw.items():
    match_found = False
    for line in lines:
        if keyword in line:
            for trigger, pattern in trigger_patterns:
                if trigger in line:
                    match = re.search(pattern, line)
                    if match:
                        evidence = match.group(1).strip()
                        is_compelling = match.group(1).strip()
                        input_ = match.group(1).strip()
                        data.append({
                            "check": keyword,
                            "input": input_,
                            "evidence": evidence,
                            "is_compelling": is_compelling
                        })
                        match_found = True
                        break
            if match_found:
                break
    if not match_found:
        data.append({
            "check": keyword,
            "input": "N/A",
            "evidence": "Not available in the log, hence no evidence",
            "is_compelling": "N/A"
        })
